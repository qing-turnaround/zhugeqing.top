func generateBoard(res []int,n int)[]string {
	ans := make([]string, 0)
	for i := 0; i < len(res); i++{
		temp := []byte{}
		for j := 0; j < n; j++ {
			if j == res[i]{
				temp = append(temp,'Q')
			}else{
				temp = append(temp,'.')
			}
		}
		ans = append(ans,string(temp))
	}
	return ans
}
