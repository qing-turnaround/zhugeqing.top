func lengthOfLIS(nums []int) int {
	//åŠ¨æ€è§„åˆ’

	n := len(nums)
	dp := make([]int, n)
	max := 0
	for i := 0; i < n; i++ {
		for j := 0; j < i; j++ {
			if nums[i] > nums[j] {
				if dp[i] < dp[j]+1 {
					dp[i] = dp[j] + 1
				}
			}
		}
		if max < dp[i] {
			max = dp[i]
		}
	}

	return max + 1

}
