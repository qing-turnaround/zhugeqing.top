func quickSort(nums []int, start, end, k int) int {

	middle := partition(nums, start, end, k)
	if middle == k {
		return nums[middle]
	} else if middle < k {
		return quickSort(nums, middle+1, end, k)
	} else {
		return quickSort(nums, start, middle-1, k)
	}
}
