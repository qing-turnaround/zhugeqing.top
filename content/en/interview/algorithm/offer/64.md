func reverseWords(s string) string {
    if len(s) == 0{
        return s
    }

    // str := []byte(s)
    n := len(s)
    //0ï¼Œ1åˆ†åˆ«è¡¨ç¤ºå•è¯ï¼Œç©ºæ ¼
    status := 0
    l,r:=0,0
    words := make([]string,0)
    for r<n{
        if status == 0{
            if s[r] == ' '{
                r++
                continue
            }else{
                status = 1
                l = r
                r++
            }
        }else{
            if s[r] == ' '{
                words = append(words,string(s[l:r]))
                status = 0
                r++
            }else{
                r++
            }
        }
    }
    if status != 0{
        words = append(words,string(s[l:r]))
    }
    
    res := ""
    for i:=len(words)-1;i>=0;i--{
        res += words[i]
        if i != 0{
            res += " "
        }
    }

    return res
}
