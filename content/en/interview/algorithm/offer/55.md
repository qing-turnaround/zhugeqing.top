func search(nums []int, target int) int {
    if len(nums) == 0{
        return 0
    }
    //äºŒåˆ†æŸ¥æ‰¾
    count := 1
    l := 0

    for  l <= r && nums[mid] != target{
        if nums[mid] < target{
            l = mid+1
        }else{
            r = mid-1
        }
        mid = (l+r)/2
    }

    if nums[mid] != target{
        return 0
    }

    //å†å‘å·¦å³æœç´¢
    left := mid-1
    right:= mid+1
    for left >=0 && nums[left] == target{

        count++
        right++
    }

    return count
}
