func findContinuousSequence(target int) [][]int {
    //æ»‘åŠ¨çª—å£
    l := 1
    r := 2
    ans := make([][]int,0)
    temp := l+r

    for l < r{
        if temp < target{
            r++
            temp += r
        }else if temp > target{
            temp -= l
            l++
        }else{
            res := make([]int,0)
            for i:=l;i<=r;i++{
                res = append(res,i)
            }

            ans = append(ans,append([]int(nil),res...))
            temp -= l
            l++
        }
    }    

    return ans
}
