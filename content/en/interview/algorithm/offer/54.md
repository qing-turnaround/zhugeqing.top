func getIntersectionNode(headA, headB *ListNode) *ListNode {
    if headA == nil || headB == nil{
        return nil
    }
    preA := headA
    preB := headB

    for preA != preB{
        if preA == nil{
            preA = headB
        }else{
            preA = preA.Next
        }
        
        if preB == nil{
            preB = headA
        }else{
            preB = preB.Next
        }
        


    }

    return preA
}
