func twoSum(nums []int, target int) []int {
    //åŒæŒ‡é’ˆ
    l := 0
    r := len(nums)-1

    for l!=r{
        temp := nums[l]+nums[r]
        if temp == target{
            return []int{nums[l],nums[r]}
        }else if temp > target{
            r--
        }else{
            l++
        }
    }

    return []int{}
}
