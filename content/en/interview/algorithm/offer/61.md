## 1.
```golang
func singleNumber(nums []int) int {
    ans := 0
    
    for i:=0;i<32;i++{
        sum := 0
        for _,v:=range nums{
            sum += v>>i & 1
        }
        if sum % 3 != 0{
            ans |= 1<<i
        }
    }

    return ans
}
```

func singleNumber(nums []int) int {
    hash := make(map[int]int)

    for i:=0;i<len(nums);i++{
        hash[nums[i]]++
    }

    for i,v :=range hash{
        if v % 3 != 0{
            return i
        }
    }

    return 0
}
