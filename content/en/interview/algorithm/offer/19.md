func isMatch(s string, p string) bool {

    s_len, p_len:= len(s),len(p)

        dp[i] = make([]bool,p_len+1)
    }   

    dp[0][0] = true

    //2.base case

        ch := p[col-1]
        if col > 1 {

                    dp[0][col] = dp[0][col-2]; 
            } else {
                    dp[0][col] = false;
            }
        } else {
            if (ch == '*') {
                dp[0][col] = true;
            }
        }
    }

    //3.çŠ¶æ€è½¬ç§»æ–¹ç¨‹
    for row:=1;row<=s_len;row++{
        ch1 := s[row-1]
        for col:=1;col<=p_len;col++{
            ch2 := p[col-1]
            if ch1 == ch2 || ch2 == '.'{
                dp[row][col] = dp[row-1][col-1]

                if col > 1{

                        dp[row][col] = true //0ä¸ªå­—ç¬¦çš„æƒ…å†µ

            }
        }
    }

    return dp[s_len][p_len]

}
