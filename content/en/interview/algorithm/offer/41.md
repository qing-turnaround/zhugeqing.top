func majorityElement(nums []int) int {
    hash := make(map[int]int)
    max := nums[0]
    hash[nums[0]]++
    n := len(nums)
    for i:=1;i<n;i++{
        hash[nums[i]]++

        if hash[nums[i]] > hash[max]{
            max = nums[i]
        }
    }

    return max
}
