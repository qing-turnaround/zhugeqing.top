func maxSubArray(nums []int) int {
    //base case
    max := nums[0]
    ans := max
    for i:=1;i<len(nums);i++{
        if max + nums[i] > nums[i]{
            max = max + nums[i]
        }else{
            max = nums[i]
        }

        if ans < max{
            ans = max
        }
    }

    return ans
}
