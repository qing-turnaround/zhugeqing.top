func treeToDoublyList(root *Node) *Node {
    if root == nil{
        return nil
    }
    
    //æ ‡è®°æœ€å°çš„èŠ‚ç‚¹
    var head *Node

    search = func(root *Node){
        if root == nil{
            return 
        }

        search(root.Left)
        if pre != nil{
            pre.Right = root
        }else{
            head = root
        }
        root.Left = pre
