func maxValue(grid [][]int) int {
    //åŠ¨æ€è§„åˆ’

    m := len(grid)
    n := len(grid[0])

    dp := make([][]int,m)
    for i:=0;i<m;i++{
        dp[i] = make([]int,n)
    }

    //å› ä¸ºåªèƒ½å‘å³å’Œä¸‹èµ°ï¼Œå¾ˆå®¹æ˜“å¾—åˆ°çŠ¶æ€è½¬ç§»æ–¹ç¨‹
    // dp[i][j] = max(dp[i-1][j],dp[i][j-1])+grid[i][j]
    
    //base case
    dp[0][0] = grid[0][0]

    for i:=0;i<m;i++{
        for j:=0;j<n;j++{
            if (i == 0 && j == 0){
                continue
            }
            if i-1 <0{
                dp[i][j] = dp[i][j-1] + grid[i][j]
            }else if j-1<0{
                dp[i][j] = dp[i-1][j] + grid[i][j]
            }else{
                dp[i][j] = max(dp[i-1][j],dp[i][j-1])+grid[i][j]
            }

        }
    }
    return dp[m-1][n-1]

}

func max(a,b int)int{
    if a > b{
        return a
    }
    return b
}
